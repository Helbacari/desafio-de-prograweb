Aplicando Fibonacci

Ahorra Fácilmente

En las finanzas personales, el ahorro lineal clásico (guardar siempre la misma cantidad fija) suele volverse monótono o insuficiente para alcanzar metas de alto impacto a corto plazo. La solución óptima a este desafío de acumulación radica en la Proporción Áurea y la Serie de Fibonacci.

Al aplicar un modelo de ahorro basado en esta secuencia, cada semana depositas la suma de lo que ahorraste en las dos semanas anteriores, multiplicado por un factor base. Gracias a la tasa de crecimiento orgánico de Fibonacci, el sistema genera una aceleración de capital exponencial moderada al principio y sumamente retadora al final, consolidando un hábito financiero inquebrantable.

🧮 Explicación del Algoritmo

El motor matemático calcula la proyección de depósitos sumando de forma iterativa los dos estados financieros anteriores. El sistema arranca con un multiplicador inicial en las semanas 1 y 2:

1+ 1→ 2→ 3→ 5→ 8→ 13→ 21→ ...

// 1. Escuchamos el clic del botón de la interfaz
document.getElementById('btn-calcular-fibonacci').addEventListener('click', calcularFibonacciEntropy);

function calcularFibonacciEntropy() {
    // 2. Captura de datos desde el DOM (Uso de const)
    const baseBits = parseFloat(document.getElementById('base-bits').value);
    const niveles = parseInt(document.getElementById('niveles').value);
    const divResultado = document.getElementById('resultado-fibonacci');

    // 3. Validación de rangos seguros
    if (isNaN(baseBits) || baseBits <= 0 || isNaN(niveles) || niveles < 2 || niveles > 24) {
        divResultado.innerHTML = "<span class='error-msg'>⚠ Parámetros inválidos.</span>";
        return;
    }

    // 4. Inicialización de variables escalares puras (Uso de let)
    let a = 1, b = 1, c;
    let html = "<div class='fib-grid'>";

    // 5. Ciclo iterativo eficiente (Template Literals modernos)
    let i = 1;
    while (i <= niveles) {
        let valorIteracion = a * baseBits;

        // Estructura limpia usando comillas invertidas, sin signos '+'
        html += `
            <div class='fib-item'>
                <span class='fib-nivel'>Iteración ${i}</span>
                <span class='fib-valor'>${valorIteracion.toFixed(0)} bits</span>
            </div>`;

        c = a + b;
        a = b;
        b = c;
        i++;
    }
    
    // 6. Renderizado final en la interfaz
    divResultado.innerHTML = `${html}</div>`;
}

Protege tu seguridad

Al igual que un candado digital inquebrantable, estos valores numéricos impiden que atacantes intercepten tus credenciales, garantizando que tu navegación e identidad se mantengan en un entorno completamente privado.

function validarCodigoPrimo() {
    const inputCodigo = document.getElementById('codigo-token');
    const divResultado = document.getElementById('resultado-primos');
    
    if (!inputCodigo || !divResultado) return;
    
    const numero = parseInt(inputCodigo.value);

    if (isNaN(numero) || numero < 1) {
        divResultado.innerHTML = `
            <div class='error-box-anim'>
                <span>⚠ ERROR CRÍTICO:</span> Por favor, ingresa un código numérico entero positivo mayor a 0.
            </div>`;
        return;
    }

    // Casos base escalares
    if (numero === 1) {
        generarGridVisual(false, "SISTEMA VULNERABLE", "El número 1 no es primo por definición.", numero);
        return;
    }
    if (numero === 2 || numero === 3) {
        generarGridVisual(true, "SISTEMA SEGURO", "Primo absoluto. Excelente nivel de protección.", numero);
        return;
    }
    if (numero % 2 === 0 || numero % 3 === 0) {
        let divisor = (numero % 2 === 0) ? 2 : 3;
        generarGridVisual(false, "SISTEMA RECHAZADO", `Código predecible. Factor directo de ${divisor}.`, numero);
        return;
    }

    // Algoritmo eficiente de primalidad pura (6k +/- 1)
    let esPrimo = true;
    let factorBuscado = 0;
    let i = 5;

    while (i * i <= numero) {
        if (numero % i === 0) {
            esPrimo = false;
            factorBuscado = i;
            break; 
        }
        if (numero % (i + 2) === 0) {
            esPrimo = false;
            factorBuscado = i + 2;
            break; 
        }
        i += 6;
    }

    if (esPrimo) {
        generarGridVisual(true, "SISTEMA SEGURO", "Código aprobado con éxito. Carece de divisores.", numero);
    } else {
        generarGridVisual(false, "SISTEMA RECHAZADO", `Ataque aritmético efectivo. Vulnerable al divisor ${factorBuscado}.`, numero);
    }
}

function generarGridVisual(valido, estadoTexto, descripcion, original) {

                <span class="fib-valor">${estadoTexto}</span>
                <p class="fib-descripcion">${descripcion}</p>
            </div>
        </div>`;
}

Aplicando Ambos

Criptografia

¿Qué sucede cuando unimos la secuencia de crecimiento de Fibonacci con las propiedades de aislamiento de los números primos? Obtenemos algoritmos de criptografía avanzada y generadores de claves pseudoaleatorias ultra-seguras.

Esta combinación genera patrones dinámicos en memoria que cambian constantemente su tamaño estructural basándose en Fibonacci, mientras que la encriptación dura se ejecuta con bases primas. Es el balance perfecto entre adaptabilidad biológica y robustez algorítmica.

// 1. Escuchamos el clic del botón de la interfaz
document.getElementById('btn-generar-token').addEventListener('click', generarTokenSimbiotico);

// Banco de datos estático (Números primos criptográficos)
const primosPool = [104729, 1299709, 15485863, 179424673];

function generarTokenSimbiotico() {
    // 2. Captura de elementos de salida del DOM
    var fibText = document.getElementById('output-fibonacci');
    var primosText = document.getElementById('output-primos');
    var tokenText = document.getElementById('output-token');

    // 3. Fase Fibonacci: Saltos orgánicos dinámicos
    var f0 = 0, f1 = 1, temp;
    var iteracionesRandom = Math.floor(Math.random() * 8) + 5; 
    
    var i = 0;
    while (i < iteracionesRandom) {
        temp = f0 + f1;
        f0 = f1;
        f1 = temp;
        i = i + 1;
    }
    fibText.innerHTML = "F(" + iteracionesRandom + ") calculado: <strong>" + f1 + "</strong>.";

    // 4. Fase Números Primos: Simulación de selección RSA
    var p = primosPool[Math.floor(Math.random() * primosPool.length)];
    var q = primosPool[Math.floor(Math.random() * primosPool.length)];
    while (q === p) { 
        q = primosPool[Math.floor(Math.random() * primosPool.length)];
    }
    
    var moduloCifradoN = p * q;
    primosText.innerHTML = "Producto compuesto (N): <strong>" + moduloCifradoN + "</strong>";

    // 5. Fusión final y formateo del Token (XXXX-XXXX-XXXX)
    var semillaFinal = moduloCifradoN + f1;
    var stringSemilla = semillaFinal.toString();
    
    var bloque1 = stringSemilla.substring(0, 4) || "8834";
    var bloque2 = stringSemilla.substring(4, 8) || "9120";
    var bloque3 = stringSemilla.substring(8, 12) || "4751";
    
    tokenText.innerText = bloque1 + "-" + bloque2 + "-" + bloque3;
}
